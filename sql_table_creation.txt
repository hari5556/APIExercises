CREATE TABLE product_inventory (id INT AUTO_INCREMENT PRIMARY KEY, BarcodeNo VARCHAR(50) NOT NULL UNIQUE, SKU VARCHAR(50) NOT NULL, Product VARCHAR(100) NOT NULL, Supplier VARCHAR(100), Style VARCHAR(50), Shade VARCHAR(50), Size VARCHAR(20), Cost DECIMAL(10,2) NOT NULL, MRP DECIMAL(10,2) NOT NULL, MOP DECIMAL(10,2), Dept VARCHAR(50), Fabric VARCHAR(50), Warehouse VARCHAR(50), WHLocation VARCHAR(50), Qty INT NOT NULL DEFAULT 0, HSNCODE VARCHAR(20), CostValue DECIMAL(12,2) GENERATED ALWAYS AS (Cost * Qty) STORED, MRPValue DECIMAL(12,2) GENERATED ALWAYS AS (MRP * Qty) STORED, MOPValue DECIMAL(12,2) GENERATED ALWAYS AS (MOP * Qty) STORED, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
